Moving objects in different directions

int frame;
int x;
int y;
void setup ()
{
background(0);
size(450,450);
frame = 1;
fill(255);
strokeWeight(10);
stroke(255);
frameRate(20);
x = 100;
y = 100;
}

void draw()
{
fill(0,0,0,20);
rect(0,0,width, height);
stroke(0,random(128,255),random(128,255));
point(x,y);
//down
if (frame >= 1 && frame <= 10)
{
y = y + 25;
}

//right
if (frame >= 10.1 && frame <= 20)
{
x = x + 25;
}

//up
if (frame >= 20.1 && frame <= 30)
{
y = y - 25;
}

//left
if (frame >= 30.1 && frame <= 40)
{
x = x - 25;
}

frame++;

if (frame > 40)
{
frame = 1;
}
}
